    <script>
        const app = {
            // --- STATE MANAGEMENT ---
            state: {
                vault: JSON.parse(localStorage.getItem('anime_os_vault_v2') || '[]'),
                sessionResults: [],
                cropper: null,
                charts: { genre: null, status: null }
            },

            // --- 1. INITIALIZATION ---
            init() {
                this.bindUI();
                this.renderVault();
                this.loadCalendar(); // Feature 6: Simulcast Calendar
            },

            bindUI() {
                const dz = document.getElementById('dropZone');
                const fi = document.getElementById('fileInput');

                dz.onclick = () => fi.click();
                dz.ondragover = e => { e.preventDefault(); dz.classList.add('dragover'); };
                dz.ondragleave = e => { e.preventDefault(); dz.classList.remove('dragover'); };
                dz.ondrop = e => { e.preventDefault(); dz.classList.remove('dragover'); this.handleFile(e.dataTransfer.files[0]); };
                fi.onchange = e => this.handleFile(e.target.files[0]);
            },

            switchTab(tabId) {
                document.querySelectorAll('.tab-section').forEach(el => el.classList.remove('active'));
                document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));
                
                document.getElementById(`tab-${tabId}`).classList.add('active');
                event.currentTarget.classList.add('active');

                if(tabId === 'stats') this.renderCharts(); // Render canvas only when visible
            },

            // --- 2. IMAGE PIPELINE (CROP & PALETTE) ---
            handleFile(file) {
                if(!file || !file.type.startsWith('image/')) return this.toast('Please upload an image file.', 'error');
                
                const reader = new FileReader();
                reader.onload = e => {
                    const img = document.getElementById('cropImage');
                    img.src = e.target.result;
                    document.getElementById('cropModal').style.display = 'flex';
                    
                    if(this.state.cropper) this.state.cropper.destroy();
                    // Slight timeout to ensure modal is rendered
                    setTimeout(() => {
                        this.state.cropper = new Cropper(img, { viewMode: 1, autoCropArea: 0.95, background: false });
                    }, 50);
                };
                reader.readAsDataURL(file);
            },

            closeModals() {
                document.getElementById('cropModal').style.display = 'none';
                document.getElementById('videoModal').style.display = 'none';
                document.getElementById('scenePlayer').pause();
                document.getElementById('scenePlayer').src = '';
                if(this.state.cropper) this.state.cropper.destroy();
            },

            // Feature 8: Aesthetic Palette Generator
            async extractPalette(canvas) {
                return new Promise(resolve => {
                    const ctx = canvas.getContext('2d');
                    // Get image data, step through pixels to find 5 dominant-ish colors
                    const data = ctx.getImageData(0, 0, canvas.width, canvas.height).data;
                    const colors = [];
                    const step = Math.floor((data.length / 4) / 5); // Take 5 distinct samples
                    
                    for(let i=0; i<5; i++) {
                        const idx = (i * step) * 4;
                        colors.push(`rgb(${data[idx]}, ${data[idx+1]}, ${data[idx+2]})`);
                    }
                    resolve(colors);
                });
            },

            // --- 3. THE CORE AI ENGINE (API INTEGRATIONS) ---
            async executeSearch() {
                const canvas = this.state.cropper.getCroppedCanvas();
                const dataUrl = canvas.toDataURL('image/jpeg', 0.9);
                
                // Get blob for API upload
                canvas.toBlob(async (blob) => {
                    this.closeModals();
                    this.showProcessing("Uploading to Trace.moe Vision Engine...");

                    try {
                        // API 1: Trace.moe
                        const formData = new FormData();
                        formData.append("image", blob);
                        const traceRes = await fetch("https://api.trace.moe/search?anilistInfo", { method: "POST", body: formData });
                        const traceData = await traceRes.json();
                        
                        if(!traceData.result || traceData.result.length === 0) throw new Error("No matching scene found in database.");
                        const match = traceData.result[0];

                        this.showProcessing("Fetching Deep Metadata from AniList...");

                        // API 2: AniList GraphQL (Features 2, 5, 11, 15)
                        const aniQuery = `
                        query ($id: Int) {
                            Media (id: $id) {
                                id title { english romaji }
                                status episodes season seasonYear bannerImage coverImage { extraLarge }
                                genres
                                characters (role: MAIN, sort: [ROLE]) { nodes { name { full } image { medium } } }
                                relations { nodes { title { english } type } }
                                recommendations (perPage: 2) { nodes { mediaRecommendation { title { english romaji } } } }
                            }
                        }`;
                        
                        const aniRes = await fetch('https://graphql.anilist.co', {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json' },
                            body: JSON.stringify({ query: aniQuery, variables: { id: match.anilist.id } })
                        });
                        const aniData = await aniRes.json();
                        const info = aniData.data.Media;

                        // API 3: Tesseract OCR (Feature 3) - Run alongside palette extraction
                        this.showProcessing("Extracting Palette & Subtitles...");
                        let ocrText = "";
                        try {
                            const tRes = await Tesseract.recognize(blob, 'jpn', { logger: m => {} });
                            ocrText = tRes.data.text.replace(/\s+/g, '').trim();
                        } catch(e) { console.warn("OCR Failed", e); }

                        const palette = await this.extractPalette(canvas);

                        // Assemble Final Object
                        const finalResult = {
                            id: Date.now().toString(),
                            aniId: info.id,
                            title: info.title.english || info.title.romaji,
                            image: dataUrl,
                            banner: info.bannerImage || info.coverImage.extraLarge || dataUrl,
                            episode: match.episode || 1,
                            totalEpisodes: info.episodes,
                            similarity: (match.similarity * 100).toFixed(1),
                            timestamp: this.formatTime(match.from),
                            video: match.video,
                            genres: info.genres || [],
                            palette: palette,
                            characters: (info.characters?.nodes || []).slice(0, 5),
                            relations: (info.relations?.nodes || []).slice(0, 2),
                            recs: (info.recommendations?.nodes || []).slice(0, 2).map(n => n.mediaRecommendation?.title?.english || n.mediaRecommendation?.title?.romaji),
                            ocr: ocrText,
                            status: 'Plan to Watch', // Feature 13 default
                            savedAt: new Date().toISOString()
                        };

                        this.state.sessionResults.unshift(finalResult);
                        this.hideProcessing();
                        document.getElementById('recentSearchHeader').style.display = 'block';
                        this.renderSearchGrid();
                        this.toast(`Identified: ${finalResult.title}`, 'success');

                    } catch (error) {
                        this.hideProcessing();
                        this.toast(error.message, 'error');
                    }
                }, 'image/jpeg');
            },

            // --- 4. RENDERERS (DOM UPDATES) ---
            renderCardHTML(item, isVault = false) {
                // Feature 9 & 11: Spoiler Protection & Total Progress
                const epTotal = item.totalEpisodes || '?';
                const epNum = item.episode || 1;
                let progressPercent = item.totalEpisodes ? (epNum / item.totalEpisodes) * 100 : 0;
                if(progressPercent > 100) progressPercent = 100;
                
                // Blur if episode is in the final 10% of the show
                const isSpoiler = item.totalEpisodes && (epNum / item.totalEpisodes) > 0.9;
                const matchClass = item.similarity >= 85 ? 'match-high' : 'match-low';

                // Stream Link (Feature 1)
                const streamLink = `https://www.crunchyroll.com/search?q=${encodeURIComponent(item.title)}`;

                return `
                <div class="anime-card">
                    <div class="card-banner-wrapper">
                        <img src="${item.banner}" class="card-banner ${isSpoiler ? 'spoiler-blur' : ''}" loading="lazy">
                        ${isSpoiler ? '<div class="spoiler-warning">⚠️ Potential Spoiler (Hover to reveal)</div>' : ''}
                    </div>
                    
                    <div class="card-content">
                        <div class="badges">
                            <span class="badge ${matchClass}">${item.similarity}% Match</span>
                            <span class="badge accent">EP ${epNum} / ${epTotal}</span>
                            <span class="badge">⏱ ${item.timestamp}</span>
                        </div>
                        
                        <div class="card-title">${item.title}</div>
                        
                        <div class="progress-container" title="${progressPercent.toFixed(0)}% into series">
                            <div class="progress-fill" style="width: ${progressPercent}%"></div>
                        </div>

                        <div class="palette-bar">
                            ${item.palette.map(c => `<div class="color-swatch" style="background:${c}" title="${c}"></div>`).join('')}
                        </div>

                        ${item.ocr ? `<div style="font-size:0.8rem; background: rgba(0,0,0,0.3); padding:0.5rem; border-left:2px solid var(--accent);">🇯🇵 ${item.ocr}</div>` : ''}

                        <div class="char-row">
                            ${item.characters.map(c => `<img src="${c.image.medium}" class="char-thumb" title="${c.name.full}">`).join('')}
                        </div>

                        ${item.relations.length ? `
                            <div class="relations-box">
                                <strong>Franchise:</strong><br>
                                ${item.relations.map(r => `• ${r.title.english || 'Unknown'} (${r.type})`).join('<br>')}
                            </div>
                        ` : ''}

                        <div class="card-actions">
                            <button class="btn" style="justify-content:center; padding:0.4rem;" onclick="app.playVideo('${item.video}')">▶ Play Scene</button>
                            <button class="btn" style="justify-content:center; padding:0.4rem;" onclick="window.open('${streamLink}', '_blank')">📺 Stream</button>
                        </div>

                        ${isVault ? `
                            <div class="vault-actions">
                                <select class="vault-select" onchange="app.updateStatus('${item.id}', this.value)">
                                    <option value="Plan to Watch" ${item.status === 'Plan to Watch' ? 'selected' : ''}>📌 Plan to Watch</option>
                                    <option value="Watching" ${item.status === 'Watching' ? 'selected' : ''}>👁️ Watching</option>
                                    <option value="Completed" ${item.status === 'Completed' ? 'selected' : ''}>✅ Completed</option>
                                </select>
                                <button class="btn btn-danger" style="padding:0.4rem;" onclick="app.deleteFromVault('${item.id}')">✖</button>
                            </div>
                        ` : `
                            <button class="btn btn-primary" style="margin-top:0.5rem; justify-content:center;" onclick="app.saveToVault('${item.id}')">💾 Save to Vault</button>
                            <button class="btn" style="margin-top:0.5rem; justify-content:center; font-size:0.8rem;" onclick="app.downloadWallpaper('${item.id}')">📥 Download Raw Frame</button>
                        `}
                    </div>
                </div>`;
            },

            renderSearchGrid() {
                document.getElementById('searchGrid').innerHTML = this.state.sessionResults.map(i => this.renderCardHTML(i, false)).join('');
            },

            renderVault() {
                const grid = document.getElementById('vaultGrid');
                if(this.state.vault.length === 0) {
                    grid.innerHTML = `<div style="grid-column: 1/-1; text-align:center; padding:3rem; color:var(--text-muted);">Your vault is empty. Search and save items here.</div>`;
                    return;
                }
                grid.innerHTML = this.state.vault.map(i => this.renderCardHTML(i, true)).join('');
            },

            // --- 5. DATA MANAGEMENT (VAULT & EXPORT) ---
            saveToVault(id) {
                const item = this.state.sessionResults.find(i => i.id === id);
                if(this.state.vault.find(i => i.aniId === item.aniId && i.episode === item.episode)) {
                    return this.toast('This exact scene is already in your vault!', 'warning');
                }
                this.state.vault.unshift(item);
                localStorage.setItem('anime_os_vault_v2', JSON.stringify(this.state.vault));
                this.renderVault();
                this.toast('Saved to Permanent Vault', 'success');
            },

            deleteFromVault(id) {
                if(!confirm("Remove this entry from your vault?")) return;
                this.state.vault = this.state.vault.filter(i => i.id !== id);
                localStorage.setItem('anime_os_vault_v2', JSON.stringify(this.state.vault));
                this.renderVault();
                if(document.getElementById('tab-stats').classList.contains('active')) this.renderCharts();
            },

            updateStatus(id, newStatus) {
                const idx = this.state.vault.findIndex(i => i.id === id);
                if(idx > -1) {
                    this.state.vault[idx].status = newStatus;
                    localStorage.setItem('anime_os_vault_v2', JSON.stringify(this.state.vault));
                    this.toast(`Status updated to ${newStatus}`, 'success');
                    if(document.getElementById('tab-stats').classList.contains('active')) this.renderCharts();
                }
            },

            wipeVault() {
                if(confirm("DANGER: This will delete your entire permanent vault. Are you sure?")) {
                    localStorage.removeItem('anime_os_vault_v2');
                    this.state.vault = [];
                    this.renderVault();
                    this.toast('OS wiped clean.', 'success');
                }
            },

            exportVault() { // Feature 4
                const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(this.state.vault, null, 2));
                const a = document.createElement('a');
                a.href = dataStr;
                a.download = `AnimeOS_Backup_${new Date().toISOString().slice(0,10)}.json`;
                a.click();
            },

            // Feature 10: Download Wallpaper
            downloadWallpaper(id) {
                const item = this.state.sessionResults.find(i => i.id === id);
                if(!item) return;
                const a = document.createElement('a');
                a.href = item.image; // Raw extracted frame
                a.download = `${item.title.replace(/[^a-z0-9]/gi, '_')}_EP${item.episode}.jpg`;
                a.click();
            },

            // --- 6. EXTERNAL APIS (CALENDAR) ---
            async loadCalendar() {
                try {
                    const res = await fetch('https://api.jikan.moe/v4/schedules?filter=today');
                    const data = await res.json();
                    
                    document.getElementById('calendarGrid').innerHTML = data.data.slice(0, 12).map(a => `
                        <div class="glass-panel" style="padding: 1rem; display:flex; gap: 1rem; align-items:center;">
                            <img src="${a.images.jpg.image_url}" style="width:60px; height:80px; object-fit:cover; border-radius:8px;">
                            <div>
                                <h4 style="margin-bottom:0.25rem;">${a.title}</h4>
                                <span class="badge accent">${a.broadcast?.string || 'TBA'}</span>
                            </div>
                        </div>
                    `).join('');
                } catch(e) {
                    document.getElementById('calendarGrid').innerHTML = "Failed to load schedule from Jikan API.";
                }
            },

            // --- 7. ANALYTICS (CHART.JS) ---
            renderCharts() { // Feature 12
                if(!this.state.vault.length) return;

                const genres = {};
                const statuses = { 'Plan to Watch': 0, 'Watching': 0, 'Completed': 0 };
                let totalEps = 0;

                this.state.vault.forEach(item => {
                    item.genres.forEach(g => genres[g] = (genres[g] || 0) + 1);
                    if(statuses[item.status] !== undefined) statuses[item.status]++;
                    totalEps += Number(item.episode) || 1; // Count episodes watched to reach this point
                });

                // Top 6 Genres
                const sortedGenres = Object.entries(genres).sort((a,b)=>b[1]-a[1]).slice(0,6);

                // Destroy old charts
                if(this.state.charts.genre) this.state.charts.genre.destroy();
                if(this.state.charts.status) this.state.charts.status.destroy();

                Chart.defaults.color = '#94a3b8';

                this.state.charts.genre = new Chart(document.getElementById('genreChart'), {
                    type: 'radar',
                    data: {
                        labels: sortedGenres.map(g => g[0]),
                        datasets: [{
                            label: 'Titles',
                            data: sortedGenres.map(g => g[1]),
                            backgroundColor: 'rgba(139, 92, 246, 0.4)',
                            borderColor: '#8b5cf6',
                            pointBackgroundColor: '#0ea5e9'
                        }]
                    },
                    options: { maintainAspectRatio: false, scales: { r: { grid: { color: 'rgba(255,255,255,0.1)' }, angleLines: { color: 'rgba(255,255,255,0.1)' } } }, plugins: { legend: { display: false } } }
                });

                this.state.charts.status = new Chart(document.getElementById('statusChart'), {
                    type: 'doughnut',
                    data: {
                        labels: Object.keys(statuses),
                        datasets: [{
                            data: Object.values(statuses),
                            backgroundColor: ['#f59e0b', '#0ea5e9', '#10b981'],
                            borderWidth: 0
                        }]
                    },
                    options: { maintainAspectRatio: false, cutout: '70%', plugins: { legend: { position: 'bottom' } } }
                });

                // Text Metrics
                const hours = (totalEps * 24) / 60; // Approx 24m per ep
                document.getElementById('metricsText').innerHTML = `
                    <div><div style="font-size:2rem; color:var(--primary); font-weight:bold;">${this.state.vault.length}</div>Total Saves</div>
                    <div><div style="font-size:2rem; color:var(--accent); font-weight:bold;">${totalEps}</div>Episodes Tracked</div>
                    <div><div style="font-size:2rem; color:var(--success); font-weight:bold;">${hours.toFixed(1)}</div>Estimated Hours</div>
                `;
            },

            // --- 8. UTILITIES ---
            playVideo(url) { // Feature 7
                if(!url) return this.toast('Video preview unavailable', 'error');
                const player = document.getElementById('scenePlayer');
                player.src = url;
                document.getElementById('videoModal').style.display = 'flex';
                player.play().catch(e => console.log("Autoplay prevented"));
            },

            formatTime(seconds) {
                if(!seconds) return "00:00";
                return new Date(seconds * 1000).toISOString().substr(14, 5);
            },

            showProcessing(text) {
                document.getElementById('processBar').style.display = 'block';
                document.getElementById('processText').innerText = text;
            },

            hideProcessing() {
                document.getElementById('processBar').style.display = 'none';
            },

            toast(msg, type='info') {
                const t = document.createElement('div');
                t.className = 'toast';
                t.style.borderColor = type === 'error' ? 'var(--danger)' : type === 'success' ? 'var(--success)' : 'var(--primary)';
                t.innerHTML = msg;
                document.getElementById('toastContainer').appendChild(t);
                setTimeout(() => { t.style.opacity = 0; setTimeout(() => t.remove(), 300); }, 3000);
            }
        };

        // Boot Application
        document.addEventListener('DOMContentLoaded', () => app.init());
    </script>
